//   char[3] :
//   char[4] :
//   char[5] :
//   char[6] :
//   char[7] :
//   char[8]   : the total number of stones in the board.
//   char[8+1] : the first stone on the board, // usually it is 7*15+7
//   char[8+2] : the second stone on the board:
//   ...
//   ...
//   char[8+totalnumberstone]: the last stone in the board.
