/* 实现对局的程序 ，计算全部N*N个格中，最应该填的格子 */
int zhineng(int a[N+1][N+1])
{
int i,j;
int k;
max.sum=-1;


for(i=0;i<=N;i++)
for(j=0;j<+N;j++)
{
w[j].sum=0;
w[j].x=i;
w[j].y=j;
}

for(i=1;i<=N-4;i++)
for(j=1;j<=N-4;j++)
{
k=zh5(i,j,a);
if(k==STOP) return (STOP);

}

for(i=1;i<=N;i++)
for(j=1;j<=N;j++)
{
if(max.sum<w[j].sum)
{
  max.sum=w[j].sum;
  max.y=i;
  max.x=j;
}
else if(max.sum==w[j].sum)
{
if(((max.y-zy)*(max.y-zy)+(max.x-zx)*(max.x-zx))>((i-zy)*(i-zy)+(j-zx)*(j-zx)))
max.sum=w[j].sum;
max.y=i;
max.x=j;
}
}
if(a[max.y][max.x]==0)
{
a[max.y][max.x]=-1;
zy=max.y;
zx=max.x;
}

}
